\begin{frame}[fragile]{Programação Modular: Ementa e Unidades de Ensino}
        \begin{block}{Ementa}
                Fatores de qualidade de software. Modularidade. Projeto Orientado para Objetos. Polimorfirsmos de Inclusão e Paramétrico. Tipos Enumeráveis e opcionais. Coleções. Tratamento de exceções. Programação Orientada a Eventos. Princípios SOLID. Padrões de Projeto. Aspectos funcionais. Teste unitário. Serialização.
        \end{block}
\end{frame}
\begin{frame}[fragile]{Programação Modular: Ementa e Unidades de Ensino}
        \begin{block}{Unidades}
                \bi
        \item 0) Nivelamento e qualidade de software 
        \item 1) POO: conceitos básicos 
        \item 2) POO e abstração 
        \item 3) Tratamento de erros e exceções 
        \item 4) Tópicos complementares 
        \item 5) Padrões de Projeto
                \ei
        \end{block}
\end{frame}
\begin{frame}[fragile]{Processo Avaliativo}
        \begin{block}{Distribuição de Pontos}
                \bi
        \item Provas: 75 pontos.
                \bi
        \item 3 x 25 pontos.*
                \ei
        \item Exercíos e trabalhos: 20 pontos
                \bi
        \item Exercíos e trabalhos teóricos e práticos: 15 pontos
        \item Prática investigativa: 5 pontos
                \ei
        \item Avalição de Desempenho Acadêmico (ADA): 5 pontos
                \ei
        \end{block}
        \begin{block}{Exercícios não pontuados}
                \bi
        \item Prática de modelagem
        \item Notação e Trechos de Código
                \ei
        \end{block}
\end{frame}
\begin{frame}[fragile]{Processo Avaliativo}
        \begin{block}{Reavaliação}
                Prova Escrita que substitui a menor nota das provas anteriores, ou seja, vale \textbf{25 pontos}.
        \end{block}
        \begin{block}{Últimas Considerações}
