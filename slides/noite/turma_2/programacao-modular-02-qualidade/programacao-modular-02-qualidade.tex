\section{Qualidade de Software}

\subsection{Engenharia de software}

\begin{frame}[fragile]{Qualidade de Software}


	Engenharia de Software busca
        
        \begin{center}
                \textit{produção de software de boa \underline{qualidade} \\ e de forma \underline{sistemática}.}
        \end{center}
        
    Fases:
    
    \bi
		\item Especificação.
        \item Desenvolvimento.
        \item Teste/manutenção.
        \item Evolução
	\ei 
        
\end{frame}

\section{Fatores de qualidade}

\begin{frame}[fragile]{Fatores de qualidade de software}
    \bi 
    
    \item Dois tipos de fatores influenciam a qualidade de um software:\\[5mm]
        \bi
            \item \textbf{Fatores externos} são aqueles com os quais os usuários interagem. \\[5mm]
            \item \textbf{Fatores internos} são aqueles com os quais os programadores interagem.
        \ei
    \ei
\end{frame}

\subsection{Fatores externos}

\begin{frame}{Fatores externos de qualidade}
    \bi 
        \item \textbf{Correção}:
        \bi
            \item Executar exatamente sua função, conforme definida pelos requisitos e especificação: garante o funcionamento no que foi previsto. \\[5mm]
        \ei
        \item \textbf{Robustez}:
        \bi
            \item Funcionar mesmo em condições anormais: garante funcionamento em situações imprevistas
            \item \textit{Graceful degradation} (degradação suave): sistema continua a operar mesmo no caso de falha de alguns dos seus componentes.
        \ei
    \ei
\end{frame}

\begin{frame}{Fatores externos de qualidade}
    \bi 
    \item \textbf{Extensibilidade}:
    \bi
    \item Medida da facilidade com que o software pode ser adaptado para atender a mudanças na sua especificação.
    \item Facilitada por projetos simples e módulos autônomos.
    \ei
    \item \textbf{Reusabilidade}:
    \bi
    \item Ser usável em novas aplicações.
    \ei
    \item \textbf{Compatibilidade}:
    \bi
    \item Medida da facilidade com que um software pode ser combinado com outros.
    \item O uso de padronização facilita a compatibilidade.
    \ei
    \ei
\end{frame}

\begin{frame}{Fatores externos de qualidade}
    \bi 
    \item \textbf{Eficiência}:
        \bi
            \item Relacionado ao desempenho. Consiste no bom uso dos recursos de hardware. \\[5mm]
        \ei
        \begin{block}{Correção $\times$ eficiência}
            Para que serve um sistema muito eficiente, mas
            que produz saídas incorretas?
        \end{block} 
        
        \vspace{5mm}
        \item Correção é recurso prioritário.

    \ei
\end{frame}


\begin{frame}{Fatores externos de qualidade}
    \bi 
    \item \textbf{Portabilidade}:
    \bi
    \item Medida da facilidade de transporte de um software
    para vários hardwares ou ambientes de programação. \\[5mm]
    \ei
    \item \textbf{Facilidade de uso}:
    \bi
    \item Preparação de dados de entrada, interpretação de
    resultados, recuperação de erros de uso, etc.
    \ei
    \ei
\end{frame}


\begin{frame}{Fatores externos de qualidade}
    \bi 
    \item \textbf{Segurança (integridade)}:
    \bi
    \item Proteção dos componentes contra acesso ou
    modificação não-autorizada. \\[5mm]
    \ei
    \item \textbf{Verificabilidade}:
    \bi
    \item Medida da facilidade de se preparar procedimentos
    de aceitação, dados para testes, etc, durante a fase
    de validação ou de operação.
    \ei
    \ei
\end{frame}

\subsection{Fatores internos}


\begin{frame}{Fatores internos de qualidade}
    \bi 
    \item Fatores externos são os percebidos pelos usuários, logo são mais  \textit{importante}.
    
    \item Mas só podem ser atingidos por meio dos fatores internos, como:
        \bi
            \item Legibilidade;
            \item Manutenibilidade;
            \item Modularidade.
        \ei
    \ei
\end{frame}



\begin{frame}{Legibilidade}
    \bi 
    \item \textbf{Legibilidade}:
        \bi
            \item Facilidade de entender o código-fonte

            \item Exemplos: nomeação de componentes, organização do código.
    
            \item \textit{Inteligibilidade}
    
            \bi
                \item Facilidade de compreender a coerência de um software
                em um nível superior do que a legibilidade proporciona.
            \ei
        \ei
    \ei
\end{frame}


\begin{frame}{Manutenibilidade}
    \bi 
    \item \textbf{Manutenibilidade}:
        \bi
            \item Facilidade para modificar um software ou corrigir problemas.
            \item Refere-se à facilidade, precisão, segurança e economia na execução de ações de manutenção
            nesse sistema ou produto.
            \item Finalidades:
            \bi
                \item Corrigir defeitos.
                \item Adequar-se a novos requisitos.
                \item Adequar-se a um ambiente novo.
            \ei
        \ei
    \ei
\end{frame}



\begin{frame}{Modularidade}
\bi
    \item Divisão do sistema em partes distintas.
    \item Ideia da ``caixa preta'': mapeamento de entradas em saídas.
    \item \textbf{Módulo}: grupo de comandos com uma funcionalidade bem definida e o mais independente possível em relação ao resto do algoritmo.
    \bi
        \item deve haver um módulo principal de controle e módulos específicos para as funções do sistema.
    \ei
\ei
\end{frame}


\begin{frame}{Modularidade}
    \bi
        \item O ideal é que os módulos não sejam grandes demais.
        \bi
            \item Facilidade de compreensão.
            \item Isolamento de função. \\[5mm]
        \ei
        \item Vantagens:
        \bi
            \item Isolamento para testes e manutenção.
            \item Interface pública uniforme.
            \item Criação de bibliotecas reutilizáveis.
            \item Economia de memória.
        \ei
    \ei
\end{frame}

\subsection{Que Fazer?}

\begin{frame}{Que Fazer?}
        \begin{block}{Dicas para escrever um bom código}
                \bi
        \item Entenda sobre o controle de versões 
        \item Utilize ferramentas de produtidade
                \bi
                  \item A IDE é sua melhor amiga
                \ei
        \item Se atentar aos erros, avisos e logs
        \item Saiba buscar ajuda 
                \bi
        \item Documentação, StackOverflow, Livros, etc
                \ei
        \item Documente seu código! 
                \bi
                  \item \textbf{Cuidado:} Documentação não substitui refatoração
                \ei
                \ei
        \end{block} 
\end{frame}
