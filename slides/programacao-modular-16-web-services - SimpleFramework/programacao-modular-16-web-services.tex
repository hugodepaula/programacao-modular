\section{Web services (WS)}

\subsection{Serviços web}


\begin{frame}{Serviços web -- \textit{web services}}

\begin{block}{Serviço web (\textit{web services})}
é uma tecnologia de chamada remota de objetos que utiliza protocolos da web (p. ex. HTTP) como meio de transporte e comunicação.
\end{block}

\begin{block}{Web services RESTful (Representational State Transfer)}
    utiliza URIs e métodos HTTP para disponibilizar acesso aos recursos da aplicação.
\end{block}

\end{frame}
 
 
\subsection{Acessando web services REST}

\section{Construindo um web service  -- framwork Simple}

\subsection{Simple framework}

\begin{frame}[fragile]{Framework Simple}

\begin{block}{Framework Simple}
    Servidor embutido em Java capaz de tratar requisições HTTP.
\end{block}

Para utilizar o framework em seu projeto:

\begin{itemize}
\item \url{http://www.simpleframework.org/}
\item Descomprimir o arquivo simple-6.0.1.jar
\item Adicionar o diretório criado no \textit{build path} do projeto.
\end{itemize}

\end{frame}


\begin{frame}[fragile]{Framework Simple}

A interface \lstinline|Container| descreve o comportamento do servidor web.

Método \lstinline|handle(Request request, Response response)| trata requisição HTTP.

\begin{lstlisting}
public class HTTPServer implements Container {

public void handle(Request request, Response response) {
    try {
        PrintStream body = response.getPrintStream();
        response.setValue("Content-Type", "text/plain");
        body.println("Olá, você requisitou: " + request.getPath());
        body.close();
    } catch (Exception e) {
        e.printStackTrace();
    }
}
...
\end{lstlisting}

\end{frame}


\begin{frame}[fragile]{Framework Simple}

Método \lstinline|main| inicia o servidor, que ficará ``ouvindo'' uma porta.

\begin{lstlisting}[basicstyle=\scriptsize]
public static void main(String[] list) throws Exception {
   int porta = 880;
    
   // Configura uma conexão soquete para o servidor HTTP.
   Container container = new HTTPServer();
   ContainerSocketProcessor servidor = 
                            new ContainerSocketProcessor(container);
   Connection conexao = new SocketConnection(servidor);
   SocketAddress endereco = new InetSocketAddress(porta);
   conexao.connect(endereco);
    
   System.out.println("Tecle ENTER para interromper o servidor...");
   System.in.read();
    
   conexao.close();
   servidor.stop();
}
}
\end{lstlisting}
\end{frame}



\begin{frame}[fragile]{Framework Simple}

\begin{block}{\scriptsize \lstinline|Container container = new HTTPServer();|}
Instancia uma classe que implementa a interface Container. Essa classe irá tratar (\textit{handle}) as requisições HTTP.
\end{block}


\begin{block}{\scriptsize \lstinline|ContainerSocketProcessor servidor = new ContainerSocketProcessor(container);|}
\textit{SocketProcessor} recebe os dados enviados pelo protocolo HTTP e os converte em um objeto \lstinline|Request|, e converte o objeto \lstinline|Response| em um fluxo de dados enviado ao cliente.
\end{block}


\begin{block}{Abrir a conexão em uma porta pré-definida}
\begin{lstlisting}[basicstyle=\scriptsize]
Connection conexao = new SocketConnection(servidor);
SocketAddress endereco = new InetSocketAddress(porta);
conexao.connect(endereco);
\end{lstlisting}
\end{block}
    
\end{frame}

\begin{frame}[fragile]{Trabalhando com dados de formulário}

\begin{lstlisting}[basicstyle=\tiny]
public void handle(Request request, Response response) {
    try {
        Query query = request.getQuery();
        PrintStream body = response.getPrintStream();
        long time = System.currentTimeMillis();
        
        response.setValue("Content-Type", "text/plain");
        response.setValue("Server", "Formulario");
        response.setDate("Date", time);
        response.setDate("Last-Modified", time);
        
        // pode ser testado com a URL: 
        // http://127.0.0.1:880/?nome=Ze;idade=32
        String nome = query.get("nome");
        int idade = query.getInteger("idade");
        
        body.println("Teste de requisição com dados de formulário.");
        body.println("NOME: " + nome + "\nIDADE: " + idade + "");
        body.close();
    } catch (Exception e) {
        e.printStackTrace();
    }
}
\end{lstlisting}

\end{frame}

\section{AJAX e JSON}

\subsection{JSON em Java}

\begin{frame}{JavaScript Object Notation}
    
    \begin{block}{JavaScript Object Notation -- JSON}
        É um formato leve para troca de informação. Baseado na linguagem JavaScript, é fácil de ser entendido pelo ser humano e pela máquina.
    \end{block}
    
    Baseado em duas estruturas:
    
    \begin{itemize}
        \item \texttt{JSONObject}: uma coleção de \texttt{pares <nome, valor>}, tais como um dicionário ou \lstinline|Map|, em Java.
        \item \texttt{JSONArray}: uma lista ordenada de valores, ou \lstinline|Array|.
    \end{itemize}
    
\end{frame}

\begin{frame}[fragile]{Exemplo: Estoque AJAX + JSON}
    
\begin{lstlisting}
function FactoryXMLHttpRequest() {
    if(window.XMLHttpRequest) {
        // Opera 8.0+, Firefox, Chrome, Safari
        return new XMLHttpRequest();
    }
    else if (window.XDomainRequest) {
        return new XDomainRequest(); // Antigo Safari
    } else if(window.ActiveXObject) {
        var msxmls = new Array(// Internet Explorer
        'Msxml2.XMLHTTP', 'Microsoft.XMLHTTP', 
        'Msxml3.XMLHTTP', 'Msxml2.XMLHTTP.7.0', 
        'Msxml2.XMLHTTP.6.0', 'Msxml2.XMLHTTP.5.0', 
        'Msxml2.XMLHTTP.4.0', 'Msxml2.XMLHTTP.3.0');
        for (var i = 0; i < msxmls.length; i++) {
            try {
                return new ActiveXObject(msxmls[i]);
            } catch (e) {      }
        }
    } else throw new Error("Could not instantiate XMLHttpRequest");
}
\end{lstlisting}
    
\end{frame}

\end{itemize}

package br.pucminas.produto_service;

public class Main {
  // Base URI the Grizzly HTTP server will listen on
  public static final String BASE_URI = 
                            "http://localhost:8080/estoque/";
\end{lstlisting}

\begin{itemize}
    \item \lstinline|BASE_URI| especifica a localização da sua aplicação. Qualquer método que você criar, será adicionado a partir dessa URI (nesse exemplo \texttt{/estoque}).
    \item O método \lstinline|HttpServer startServer()| instancia o servidor web.

\begin{frame}[fragile]{Código padrão da Aplicação RESTful -- Jersey/Grizzly}

\lstinline|ResourceConfig| define o pacote em que serão armazenadas as classes que irão tratar as requisições HTTP.

\begin{lstlisting}
public static HttpServer startServer() {
  // create a resource config that scans for JAX-RS resources
  // and providers in br.pucminas.ws_03_produto_service package
  final ResourceConfig rc = new
       ResourceConfig().packages("br.pucminas.ws_03_produto_service");
 
  // create and start a new instance of grizzly http server
  // exposing the Jersey application at BASE_URI
  return GrizzlyHttpServerFactory
            .createHttpServer(URI.create(BASE_URI), rc);
}
\end{lstlisting}
    
\end{frame}

\subsection{Rotas com \textit{annotations}}


\begin{frame}{Métodos (ou verbos) de requisição HTTP}

\begin{itemize}
    \item Principais métodos HTTP que podem ser usados em um serviço web.
\end{itemize}    


\begin{center}
\begin{tabular}{p{1.3cm}p{.8\linewidth}}
\hline
\textbf{GET}    & Solicita um recurso. Pode passar parâmetro na própria URI. \\
\textbf{POST}   & Cria um recurso. Dados encaminhados no corpo da requisição, e não na URI. \\
\textbf{PUT}    & Atualiza um recurso caso já exista ou cria se não existir.\\
\textbf{DELETE} & Exclui um recurso.\\
\hline
\end{tabular}   
\end{center}
\end{frame}


\begin{frame}{Roteamento de requisições}
\begin{itemize}
    \item é importante definir padrões para a criação de URIs.
    \item  Em uma API REST, cada modelo (recurso que se deseja persistir), cria um conjunto de rotas associado às operações CRUD de armazenamento de dados.
\end{itemize}

\begin{center}
\begin{tabular}{ll}
\hline
\textbf{CRUD} 	  & \textbf{REST (Método / URI)} \\ \hline
Create            & PUT    /modelName  \\ 
                  & POST   /modelName \\
Read (Retrieve)   & GET    /modelName?filter=...  \\
Update (Modify)   &	PUT    /modelName 	\\
Delete (Destroy)  & DELETE /modelName/modelID \\
\hline
\end{tabular}
\end{center}
\end{frame}

\begin{frame}[fragile]{Criando rotas com annotations}
    \begin{itemize}
        \item Cada recurso deve ter uma rota raiz (\lstinline|@Path|): \\
        
        \begin{lstlisting}

\begin{frame}{CORS Filter}

\begin{block}{Cross-Origin Resource Sharing (CORS)}
    Mecanismo de segurança do HTTP que utiliza o cabeçalho de uma página (\textit{head}) para avisar que a aplicação está executando em uma origem.
\end{block}

Uma aplicação executa uma \textbf{cross-origin HTTP request} quando solicita recursos de outra origem. 
\end{frame}

\begin{frame}[fragile]{CORS Filter}
\begin{lstlisting}
@Provider
public class CorsFilter implements ContainerResponseFilter {

  @Override
  public void filter(final ContainerRequestContext requestContext, 
            final ContainerResponseContext cres) throws IOException {
    cres.getHeaders().add("Access-Control-Allow-Origin", "*");
    cres.getHeaders().add("Access-Control-Allow-Headers", 
                      "origin, content-type, accept, authorization");
    cres.getHeaders().add("Access-Control-Allow-Credentials", "true");
    cres.getHeaders().add("Access-Control-Allow-Methods", 
                      "GET, POST, PUT, DELETE, OPTIONS, HEAD");
    cres.getHeaders().add("Access-Control-Max-Age", "1209600");
  }

}
