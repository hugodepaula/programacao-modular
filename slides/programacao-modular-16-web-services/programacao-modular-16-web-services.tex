\section{Web services (WS)}

\subsection{Serviços web}


\begin{frame}{Serviços web -- \textit{web services}}

\begin{block}{Serviço web (\textit{web services})}
é uma tecnologia de chamada remota de objetos que utiliza protocolos da web (p. ex. HTTP) como meio de transporte e comunicação.
\end{block}

\begin{block}{Web services RESTful (\textit{Representational State Transfer})}
    utiliza URIs e métodos HTTP para disponibilizar acesso aos recursos da aplicação.
\end{block}

\end{frame}
 
 
\subsection{Acessando web services REST}

\begin{frame}[fragile]{Acessando web services REST}

Um serviço web REST disponibiliza dados ou recursos a partir de uma URI.

\begin{block}{\small Resultado do acesso a \url{www.thomas-bayer.com/sqlrest/}}
\begin{lstlisting}[language=HTML,basicstyle=\tiny]
<resource xmlns:xlink="http://www.w3.org/1999/xlink">
 <CUSTOMERList xlink:href="http://(...)/sqlrest/CUSTOMER/">CUSTOMER</CUSTOMERList>
 <INVOICEList xlink:href="http://(...)/sqlrest/INVOICE/">INVOICE</INVOICEList>
 <ITEMList xlink:href="http://(...)/sqlrest/ITEM/">ITEM</ITEMList>
 <PRODUCTList xlink:href="http://(...)/sqlrest/PRODUCT/">PRODUCT</PRODUCTList>
</resource>
\end{lstlisting}
\end{block}

\end{frame}

\begin{frame}[fragile]{Acessando web services REST}
\begin{itemize}
\item O exemplo indica que existem 4 web services disponíveis:

\begin{enumerate}
    \item \textbf{CUSTOMER}: dados de cliente.

    \item \textbf{ITEM}: dados do item do pedido.
    \item \textbf{PRODUCT}: dados do produto.
\end{enumerate}

\end{itemize}

\end{frame}

\subsection{Spark framework}

\begin{frame}[fragile]{Spark Framework}

\begin{block}{Spark framework}
    é um micro framework para a criação de aplicações web a partir do Java 8 (ou Kotlin). Disponível em: \url{http://sparkjava.com/}.
\end{block}

Para utilizar o framework em seu projeto você deve utilizar o gerenciador de dependências chamado Maven (\url{https://maven.apache.org/}).

\end{frame}

\begin{frame}[fragile]{Configuração do Maven}
	\begin{itemize}
		\item Maven é uma ferramenta de gerenciamento de projeto.
		\item Controla configuração e dependências.
		\item Instalação: \url{https://maven.apache.org/download.cgi}
		\begin{itemize}
			\item Baixar e descomprimir o arquivo zip.
			\item Adicionar o diretório \textbf{bin} ao \textbf{PATH}.
		\end{itemize}
		\item É necessário que a variável \textbf{JAVA\_HOME} esteja configurada.
	\end{itemize}
\end{frame}

\begin{frame}[fragile]{Métodos HTTP}

\begin{lstlisting}[basicstyle=\tiny]
public static void main(String[] args) {
   port(6789);
   get("/hello", (request, response) -> "<h1>Nosso primeiro Web Server!</h1>");

   post("/hello", (request, response) -> "Web Server recebeu: " + request.body());

   get("/private", (request, response) -> { response.status(401); 
                                            return "Área privada!!!"; });

   get("/users/:name", (request, response) -> "Usuário selecionado: " + 
                                               request.params(":name"));

   get("/news/:section", (request, response) -> { response.type("text/xml");
                           return "<?xml version=\"1.0\" encoding=\"UTF-8\"?><news>" + 
                                  request.params("section") + "</news>";
   });

   get("/protected", (request, response) -> {
       halt(403, "I don't think so!!!");
       return null; });

   get("/redirect", (request, response) -> { response.redirect("/news/world");
                                             return null; });

   get("/", (request, response) -> "root");
}
\end{lstlisting}

\end{frame}


\subsection{Roteamento de requisições}

\begin{frame}[fragile]{Roteamento de requisições}
\begin{itemize}
\item Uma rota é formada por três elementos:
\begin{itemize}
	\item verbo: get, post, put, delete, head, trace, connect, options
	\item caminho: hello, /users/:name
	\item \textit{callback}: (request, response) -> \{ \}
\end{itemize}
\end{itemize}

\begin{lstlisting}[basicstyle=\scriptsize]
get("/", (request, response) -> {      // Exibe alguma coisa    
});

post("/", (request, response) -> {     // Cria alguma coisa     
});

put("/", (request, response) -> {      // Atualiza alguma coisa 
});

delete("/", (request, response) -> {   // Apaga alguma coisa
});

options("/", (request, response) -> {  // Verifica alguma coisa
});
\end{lstlisting}


\end{frame}


\begin{frame}[fragile]{Roteamento de requisições: CRUD}
	\begin{itemize}
		\item Importante definir uma tabela de rotas antes de começar a codificação.
		\item Uma tabela padrão de CRUD (criar, consultar, atualizar e apagar) segue o seguinte padrão:
	\end{itemize}
	
\begin{tabular}{lll}
	\hline
\bf verbo & \bf path & \bf descrição \\ \hline
post & /produto      & adiciona um novo produto no Estoque \\
get  & /produto/:id  & recupera os dados do produto :id \\
put  & /produto/:id  & atualiza os dados do produto :id \\
delete  & /produto/:id  & remove o produto :id \\
get  & /produtos      & recupera toda a lista de produtos \\

\hline
\end{tabular}	
	
\end{frame}

\begin{frame}[fragile]{Disponibilização de arquivos estáticos}
	\begin{itemize}
		\item Os arquivos de \textit{front-end} (.html) são considerados arquivos estáticos.
		
		\item Eles devem ser disponibilizados para poderem ser acessados a partir do servidor.

		\item O comando \lstinline|staticFileLocation("/public");| avisa que os arquivos estáticos estarão disponíveis em uma pasta \verb|public|, a partir da base do projeto. NO caso do Maven, a base do projeto é o diretório \verb|"/main/resources"|.
		
	\end{itemize}
\end{frame}

\section{JSON e Redirecionamento}

\subsection{JSON em Java}

\begin{frame}{JavaScript Object Notation}
    
    \begin{block}{JavaScript Object Notation -- JSON}
        É um formato leve para troca de informação. Baseado na linguagem JavaScript, é fácil de ser entendido pelo ser humano e pela máquina.
    \end{block}
    
    Baseado em duas estruturas:
    
    \begin{itemize}
        \item \texttt{JSONObject}: uma coleção de \texttt{pares <nome, valor>}, tais como um dicionário ou \lstinline|Map|, em Java.
        \item \texttt{JSONArray}: uma lista ordenada de valores, ou \texttt{Array}.
    \end{itemize}
    
\end{frame}

\begin{frame}[fragile]{Mapeamento entre JSON e entidades Java}
    
\begin{tabular}{cc}

\subsection{Redirecionamento}



\begin{frame}{Redirecionamento}
	
	\begin{itemize}
		\item Um redirecionamento ocorre quando uma requisição é redirecionada para outro endereço.
		\item Isto é útil quando Se deseja criar um fluxo de páginas entre as requisições.
		\item Por exemplo, se um produto não for encontrado, pode-se desejar redirecionar a requisição para uma tela de erro, ou voltar para a tela de cadastro.
		\item No Spark Java, isso é feito pelo código: \lstinline|response.redirect("/novo");|.
		\item Existem outras formas de redirecionamento baseado em métodos HTTP. Veja a documentação para mais detalhes.
	\end{itemize}
	
\end{frame}
