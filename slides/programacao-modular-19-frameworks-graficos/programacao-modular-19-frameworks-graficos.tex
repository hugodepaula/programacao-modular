\section{Java GUI Toolkits}

\subsection{Java Foundation Classes} 

\begin{frame}{\textit{Java Foundation Classes} (JFC)}

Conjunto de pacotes para desenvolvimento de aplicações gráficas baseadas em formulário.

\begin{itemize}
\item \textbf{Componentes}: também chamados de \textit{widgets}, são classes com representação gráfica (renderizáveis). Por exemplo: botões, campos, menus, janelas, etc.

\item \textbf{Gerenciadores de Layouts}: são classes facilitam o gerenciamento da posição e tamanho dos componentes na tela. 

\item \textbf{Eventos}: representam ações realizadas sobre os componentes que provocam uma reação por parte dos tratadores de eventos. Podem ser entradas do usuário ou ações realizadas pelo sistema.
\end{itemize}


\end{frame}

\begin{frame}{\textit{Java Foundation Classes} (JFC)}

Pacotes para construção de interfaces gráficas com o usuário (GUI).

\begin{block}{\textit{Abstract Windowing Toolkit} (AWT)}
Provê um \textit{toolkit} de componentes gráficos que funcionam como uma camada sobre a GUI fornecida pelo sistema operacional nativo. \textit{Heavyweight}.
\end{block}

\begin{block}{\textit{Swing}}
Provê conjunto de componentes gráficos mais avançados, que suportam \textit{look\&feel} plugáveis. São 100\% Java, e independem do sistema operacional nativo. \textit{Lightweight}.
\end{block}

\end{frame}

\begin{frame}{Outros \textit{toolkits} gráficos}

\begin{block}{\textit{Standard Widget Toolkit} (SWT)}
Desenvolvido originalmente pela IBM, hoje é \textit{open-source} parte do Eclipse. São \textit{Heavyweight}, mas estendem a funcionalidade do AWT e são focados em desempenho.
\end{block}

\begin{block}{JavaFX}
Originalmente para criar \textit{Rich Internet and Mobile Applications}. Se consolidou como um \textit{toolkit} para interfaces gráficas no JRE/JDK 8.
\end{block}

\end{frame}

\section{Swing}

\subsection{Components e Containers}

\begin{frame}{Components e Containers}

\begin{itemize}
\item Components:  usados para construir a interface. Todo componente visual é um \lstinline|Component| no AWT ou \lstinline|JComponent| no Swing.

\item Containers: Capazes de armazenar uma lista de Components. São a base para construção de painéis e janelas (\textit{Frames}).

\end{itemize}
\end{frame}

\begin{frame}[fragile]{Criando uma janela}
\begin{lstlisting}
public class PrimeiroFrame {

 public static void main(String[] args) {
  JFrame janela = new JFrame("Meu primeiro frame");

  janela.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);

  Container contentPane = janela.getContentPane();

  JLabel label = new JLabel(
       "Eu sou um rótulo (componente \"JLabel\") e exibo um texto.");
  label.setForeground(Color.WHITE);
  label.setBackground(Color.BLACK);
  label.setOpaque(true);
  contentPane.add(label);

  janela.pack();
  janela.setVisible(true);
 }
}

\subsection{Menus}
\begin{frame}{Menus}
\begin{itemize}
\item \lstinline|JMenuBar|: barra de menu onde serão adicionados os menus.

\item \lstinline|JMenu|: menus \textit{pull-down}, como Arquivo, Editar, etc.

\item \lstinline|JMenuItem|: itens de menu adicionados nos menus individuais, tais como Abrir, Salvar, etc.
\end{itemize}
\end{frame}
