\section{SOLID}


\begin{frame}{\textit{SOLID Principles}}

Proposto por Robert Martin\footnote{\textit{\tiny Martin, Robert Cecil (2002). Agile software development: principles, patterns, and practices. Upper Saddle River, NJ: Pearson Education. ISBN 9780135974445.}}: 
\begin{itemize}
    \item \textit{\textbf{S}ingle responsibility}
    \item \textit{\textbf{O}pen/Closed}
    \item \textit{\textbf{L}iskov substitution}
    \item \textit{\textbf{I}nterface segregation}
    \item \textit{\textbf{D}ependency inversion}
\end{itemize}
\end{frame}

\subsection{Single responsibility principle (SRP)}

\begin{frame}{\textit{Single responsibility principle (SRP)}}

\begin{block}{\textit{Single responsibility principle (SRP)}}
``Uma classe deve ter apenas um motivo para mudar, o que significa que ela deve ter uma única função.''
\end{block}

\begin{itemize}
	\item ``Responsabilidade'': motivo para mudança.
	\item Obtida com coesão e encapsulamento.
\end{itemize}

\begin{block}{\footnotesize Roger Whitney, 2016, \textit{Advanced Object-Oriented Design \& Programming} }
Tente descrever uma classe com 25 palavras ou menos, e não use ``e'' ou ``ou''. Se você não consegue fazer isso, pode ser que você tenha mais de uma classe.
\end{block}
\end{frame}

\begin{frame}{\textit{Single responsibility principle (SRP)}}

Considere uma classe que realiza as seguintes operações:

\begin{block}{}
    \begin{enumerate}
        \item Abrir uma conexão com o banco de dados.
        \item Realizar consulta ao banco de dados.
        \item Escrever esses dados em um arquivo.
    \end{enumerate}
\end{block}


Quais seriam seus "motivos para mudança":

\begin{block}{}
    \begin{enumerate}
        \item E se for utilizado um novo banco de dados?
        \item E se for utilizado um ORM para gerenciar as consultas?
        \item E se a estrutura dos dados de escrita for alterada?
    \end{enumerate}
\end{block}

Cada mudança pode afetar as demais operações. O ideal seria ter três classes distintas, cada uma com sua responsabilidade.

\end{frame}

	\end{itemize}

\subsection{Open/closed principle (OCP)}

\begin{frame}{\textit{Open/closed principle (OCP)}}

\begin{block}{\textit{Open/closed principle (OCP)}}
``Objetos ou entidades devem ser abertos para extensão, mas fechados para modificação.''
\end{block}


\begin{itemize}
	\item Uma classe deve ser facilmente extensível, sem a necessidade de modificação.
	\item Mudanças apenas para correção de código.
	\item Extensão: classes abstratas e interfaces.
\end{itemize}
\end{frame}

public abstract class Arquivo {

}

public class Docx extends Arquivo {

   public void abrirDocx() { }
   public void fecharDocx() { }
   public void salvarDocx() { }

public class Txt extends Arquivo {

public class GerenciadorDeArquivos {
   
   public salvartodos(List<Arquivo> arquivos) {
      for (Arquivo arquivo : arquivos) {
         if (arquivo instanceof Docx)

         else if (arquivo instanceof Txt)

      }
   }
}

\begin{frame}{\textit{Open/closed principle (OCP)}}
    \begin{block}{Ideia geral do OCP}
        Se a classe \lstinline|Shulambs| foi escrita pelo desenvolvedor A, e o desenvolvedor B precisa de alguma modificação nessa classe, ele deveria conseguir fazer isso facilmente estendendo a classe \lstinline|Shulambs|, mas não modificando a mesma.
    \end{block}
\end{frame}

\subsection{Liskov substitution principle (LSP)}

\begin{frame}{\textit{Liskov substitution principle (LSP)}}

\begin{block}{\textit{Liskov substitution principle (LSP)}}
Se $q(x)$ é uma propriedade demonstrável dos objetos $x$ de tipo $T$. Então $q(y)$ deve ser verdadeiro para objetos $y$ de tipo $S$, onde $S$ é um subtipo de $T$\footnote{\textit{\tiny LISKOV, Barbara: Data abstraction and hierarchy, 1987}}.
\end{block}

\begin{itemize}
\item Objetos que fazem parte de um programa podem ser substituídos por instâncias de seus subtipos sem prejuízo para a correção do programa.
\end{itemize}
\end{frame}


\begin{frame}[fragile]{\textit{Liskov substitution principle (LSP)}}
\begin{tabular}{ll}
\begin{minipage}{.68\textwidth}
\begin{itemize}
\item Violação comum do LSP: verificação de tipo em tempo de execução.
\item Violação sutil: uso da herança desconsiderando o comportamento.
	\begin{itemize}
	\item Ex: \texttt{Quadrado} \textit{é um tipo de} \texttt{Retângulo}.
    \end{itemize}
\end{itemize}
\end{minipage} & 
\begin{minipage}{.3\textwidth}
    \begin{center}
    \begin{tikzpicture} 
     \umlemptyclass[font=\fontsize{7}{7}\selectfont]{Retangulo}
     \umlemptyclass[font=\fontsize{7}{7}\selectfont,y=-2]{Quadrado}
     \umlinherit{Quadrado}{Retangulo}
    \end{tikzpicture}
    \end{center}
\end{minipage} \\
\end{tabular}
\end{frame}

\begin{frame}[fragile]{\textit{Liskov substitution principle (LSP)}}
\begin{multicols}{2}
\begin{lstlisting}[basicstyle=\tiny]
public class Retangulo {

  private double altura;
  private double largura;
  
  public double getArea() { 
    return altura * largura; 
  }
  
  public void setAltura(double altura) {
    if (altura > 0) 
      this.altura = altura;
  }
  
  public void setLargura(double largura) {
    if (largura > 0) 
      this.largura = largura;
  }
}
public class Quadrado extends Retangulo {

  @Override
  public void setAltura(double altura) {
      super.setLargura(altura);
      super.setAltura(altura);
  }
  
  
  @Override
  public void setLargura(double largura) {
      super.setLargura(largura);
      super.setAltura(largura);
  }
  
  
}


\end{lstlisting}
\end{multicols}
\end{frame}


\begin{frame}[fragile]{\textit{Liskov substitution principle (LSP)}}
\begin{lstlisting}[basicstyle=\tiny]
public class Aplicacao {
   public static void main(String[] args) {
   
      Retangulo ret = new Retangulo();
      ret.setaltura(3);
      ret.setLargura(2);
      
      if (ret.getArea() != 6) {
         System.out.println("Algo está errado!");
      }
      
      ret = new Quadrado();
      ret.setaltura(3);
      ret.setLargura(2);
      
      if (ret.getArea() != 6) {
         System.out.println("Algo está errado!");
      }
   }
}
\end{lstlisting}

\textbf{Em resumo}: 
Herança deve considerar a relação \textit{``é-um''} para o comportamento do objeto.
\end{frame}


\subsection{Interface segregation principle (ISP)}

\begin{frame}{\textit{Interface segregation principle (ISP)}}

\begin{block}{\textit{Interface segregation principle (ISP)}}
Clientes não devem ser forçados a implementar uma interface que não usam, ou seja, não devem ser forçados a depender de métodos que não usam.
\end{block}

\begin{itemize}
\item O acúmulo de funções em uma interface pode torná-la não-coesa e forçar comportamentos não comuns a uma classe.
\item Exemplo: 
\begin{itemize}
	\item Uma \texttt{Porta} e um \texttt{Temporizador} (\textit{timer}).
	\item Uma porta de geladeira que deve disparar um alarme se ficar mais de 20s aberta.
\end{itemize}
\end{itemize}
\end{frame}

\subsection{Dependency inversion principle (DIP)}

\begin{frame}{\textit{Dependency inversion principle (DIP)}}

\begin{block}{\textit{Dependency inversion principle (DIP)}}
``Módulos de alto nível não devem depender de módulos de baixo nível. Ambos devem depender de abstrações; Abstrações não devem depender de detalhes. Detalhes devem depender de abstrações.''.
\end{block}

\begin{itemize}
    \item Nenhuma variável deve conter referência para uma classe concreta.
    \item Nenhuma classe deve derivar de uma classe concreta.
    \item Nenhum método deve sobrescrever métodos implementados em sua classe base.
\end{itemize}
\end{frame}
