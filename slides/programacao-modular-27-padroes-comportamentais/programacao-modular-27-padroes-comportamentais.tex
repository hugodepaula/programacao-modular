\section{Padrões Comportamentais}

\begin{frame}{Padrões de projeto comportamentais}

\begin{block}{Padrões Comportamentais}
Organizam a comunicação entre objetos de forma clara e eficiente.
\end{block}

\begin{itemize}
\item Exemplos mais comuns: Chain of Responsibility, Observer (Model-View-Controller), iterator, visitor, template e strategy.
\item Exemplos menos comuns: Interpreter e Mediator.


\end{itemize}

\end{frame}

\section{Chain of Responsibility}

\begin{frame}{Chain of Responsibility}

\begin{block}{Chain of Responsibility}
Descreve uma forma de organizar os objetos em cadeia (como na forma de uma corrente).
\end{block}

\begin{itemize}
\item Útil quando alguns comandos devem ser tratados por diferentes objetos, cada um deles passando o comando ao próximo objeto na forma de uma corrente.

\item Cada receptor de uma requisição contém uma referência a outro receptor.
\end{itemize}
\end{frame}

\begin{frame}[fragile]{Chain of Responsibility: exemplo Logger}

\begin{lstlisting}
public abstract class Logger {
  public static int INFO = 1;
  public static int DEBUG = 2;
  public static int ERRO = 3;
  protected int nivel;
  // proximo elemento na cadeia de responsabilidades
  protected Logger next;

  public void setNext(Logger next) {
    this.next = next;
  }

  public void log(int nivel, String msg) {
    if (this.nivel <= nivel) {
      escrever(mensagem);
    }
    if (proximoLogger != null) {
      proximoLogger.log(nivel, msg);
    }
  }
  abstract protected void write(String msg);
}
\end{lstlisting}
\end{frame}


\begin{frame}[fragile]{Chain of Responsibility: exemplo Logger}

\begin{lstlisting}[basicstyle=\tiny]
public class LoggerConsole extends Logger {
  public LoggerConsole(int nivel) {
    this.nivel = nivel;
  }
  @Override
  protected void write(String msg) {
    System.out.println("Console Padrao::Logger: " + msg);
  }
}

public class LoggerErro extends Logger {
  public LoggerErro(int nivel) {
    this.nivel = nivel;
  }
  @Override
  protected void write(String msg) {
    System.out.println("Console de Error::Logger: " + msg);
  }
}

public class LoggerArquivo extends Logger {
  public LoggerArquivo(int nivel) {
    this.nivel = nivel;
  }
  @Override
  protected void write(String msg) {
    System.out.println("Arquivo::Logger: " + msg);
  }
}
\end{lstlisting}
\end{frame}



\begin{frame}[fragile]{Chain of Responsibility: exemplo Logger}

\begin{lstlisting}

public class Cliente {

  private static Logger getCadeiaDeLoggers() {
    Logger errorLogger = new LoggerErro(Logger.ERRO);
    Logger fileLogger = new LoggerArquivo(Logger.DEBUG);
    Logger consoleLogger = new LoggerConsole(Logger.INFO);
    errorLogger.setNext(fileLogger);
    fileLogger.setNext(consoleLogger);
    return errorLogger;
  }

  public static void main(String[] args) {
    Logger loggerChain = getCadeiaDeLoggers();
    loggerChain.log(Logger.INFO, "Esta é uma informação de evento.");
    loggerChain.log(Logger.DEBUG, "Esta é uma informação de Debug.");
    loggerChain.log(Logger.ERRO, "Esta é uma mensagem de erro.");
  }
}
\end{lstlisting}
\end{frame}

\section{Observer}

\begin{frame}{Observer}

    \begin{block}{Observer}
        Relação de um-para-muitos entre objetos de tal forma que quando um objeto é modificado os objetos dependentes são notificados automaticamente.
    \end{block}
\end{frame}


\subsection{Solução geral}

\begin{frame}{Observer: solução geral}
    \begin{center}\tiny
        \begin{tikzpicture}
        \umlclass[type=abstract, font=\fontsize{8}{8}\selectfont]{Observer}{-- subject : Subject}{+ \textit{notify()}}
        \umlclass[font=\fontsize{8}{8}\selectfont,y=-2]{ConcreteObserver}{}{+ notify()}
        \umlclass[font=\fontsize{8}{8}\selectfont,x=5,y=-0.7]{Subject}{-- observers : List<Observer> \\ -- state : State }{ + getState() : State \\ + setState(: State) : void \\ + attach(: Observer) : void \\ + notifyAllObservers() : void}
        \umlVHVinherit{ConcreteObserver}{Observer}
        \umluniassoc[font=\fontsize{8}{8}\selectfont,arg2=notify,pos2=0.5]{Subject}{Observer}
        \end{tikzpicture} \\[5mm]
    \end{center}

    \begin{itemize}
        \item \textbf{Subject}: permite a conexão de observadores e possui o estado que será observado.
        \item \textbf{Observer}: se conecta a um Subject e declara o método que é executado no momento da notificação.
        \item \textbf{ConcreteObserver}: implementa o método de notificação.
    \end{itemize}

\end{frame}
